# A. Linear Search Function
def linear_search(arr, target):
    for i in range(len(arr)):
        if arr[i] == target:
            return i
    return -1

arr = [10, 20, 30, 40]
print(linear_search(arr, 40))

# B. Find Value in DataFrame
import pandas as pd

def find_df(df, column_name, value):
    return df.index[df[column_name] == value].tolist()

df = {"Name": ["ABC", "XYZ", "PQR"], "Age": [20, 25, 25]}
df = pd.DataFrame(df)

indices = find_df(df, "Age", 25)
if indices:
    print(f"{indices}")
else:
    print("Not found")

# C. Find Substring in File
def find_substring(file_name, substring):
    occurrences = []
    with open(file_name, 'r') as file:
        lines = file.readlines()
        for i, line in enumerate(lines):
            if substring in line:
                occurrences.append(i + 1)
    return occurrences

x = find_substring("f.txt", "Hello")
print(x)

# D. Search in NumPy Array
import numpy as np

def np_arr(np_array, element):
    for i in range(len(np_array)):
        if np_array[i] == element:
            return i
    return -1

np_array = np.array([10, 20, 30, 40])
x = np_arr(np_array, 20)
print(x)
